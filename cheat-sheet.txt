# Crear nuevo proyecto
ng new [nombre del proyecto]

# Levantar la app (y abrirla en el navegador)
ng serve --open
ng serve -o

# Crear componente automaticamente con todos sus archivos y lo registra
ng generate component [nombre del componente]
ng g c [nombre del componente]

ng g c [directorio]/[nombre del componente] --flat #no crea una carpeta sino que lo incorpora
